# 開始測試

# 測試版本

  • vue 2.6.10
  • pagemap 1.2.0

# 專案建置與套件安裝

  • 透過 vue cli 建立新的 vue 專案
> vue create pagemap
  • 安裝 Pagemap 套件
> yarn add pagemap // or npm install pagemap
  • 執行專案
> yarn serve // or npm run serve

# 加入 Pagemap 套件

  • 開啟 App.vue
  • 加入測試碼

template

<template>
  <div id="app">
    <img alt="Vue logo" src="./assets/logo.png">
    <HelloWorld msg="Welcome to Your Vue.js App"/>
    <!-- 加入 canvas -->
    <canvas id='map'></canvas>
  </div>
</template>

script

<script>
import HelloWorld from './components/HelloWorld.vue'
// 載入 pagemap 套件
import pagemap from 'pagemap/dist/pagemap'

export default {
  name: 'app',
  components: {
    HelloWorld
  },
  mounted(){
      // 設定 pagemap 元件
    pagemap(this.$el.querySelector('#map'), {
            // viewport: document.querySelector('#container'),
            interval: 50,
            styles: { div: 'rgba(0,0,0,0.1)' },
            back: 'rgba(255,255,255,1)'
        })
  }
}
</script>

style

#app {
  font-family: 'Avenir', Helvetica, Arial, sans-serif;
  -webkit-font-smoothing: antialiased;
  -moz-osx-font-smoothing: grayscale;
  text-align: center;
  color: #2c3e50;
  margin-top: 60px;
  /* 故意設定頁面尺寸超大 */
  width: 2000px;
  height: 2000px;
  
}
/*  設定 minimap 位置為右上角 */
#map {
    position: fixed;
    top: 0;
    right: 0;
    width: 200px;
    height: 100%;
    z-index: 100;
}
