# 開始測試

# 測試版本

  • vue 2.6.10
  • bowser 2.7.0

# 專案建置與套件安裝

  • 透過 vue cli 建立新的 vue 專案
> vue create bowser
  • 安裝 bowser 套件
> yarn add bowser // or npm install bowser
  • 執行專案
> yarn serve // or npm run serve

# 加入 bowser 套件

  • 開啟 App.vue
  • 加入測試碼

script

import HelloWorld from './components/HelloWorld.vue'

// 載入 bowser 套件
import bowser from 'bowser'

export default {
  name: 'app',
  components: {
    HelloWorld
  },
  mounted() {
    // 使用 bowser 讀取瀏覽器資訊
   console.log(bowser.getParser(window.navigator.userAgent));
  }
}
